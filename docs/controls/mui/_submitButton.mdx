> a submit button that displays a loading indicator while the form is submitting<br />
> **Render **[`<Button type="submit" />`](https://mui.com/components/buttons/)

:::info
When submitting, this button will always be disabled. [`disableWhileSubmitting`](/docs/Form/react-hook-form/api#props) has no effect on this control. If the form contains errors, the button is also disabled.
:::

<Props>
  <Prop icon="cf.png" name="displayLoading" type="boolean" defaultValue="true">
    If <code>true</code>, a loading indicator is added to the label.<br/>
    see <code>loadingComponent</code> below to customize what is appended.
  </Prop>

  <Prop icon="cf.png" name="loadingComponent" type="node" defaultValue="<CircularProgressÂ />">
    Loading indicator to append to the label when the form is submitting. (default: <a href="https://mui.com/components/progress/#circular" target="_blank"><code> {"<CircularProgress />"} </code></a>)
  </Prop>

  <Prop icon="cf.png" name="alternateLoadingContent" type="node">
    If provided, this replaces the <strong>whole</strong> label of the button when submitting.
  </Prop>

  <Prop icon="mui.png" name="color" type="string" defaultValue='"primary"'>
    <a href="https://mui.com/api/button/#props" target="_blank">Material-UI <code> color </code></a> property <strong>defaulted to "primary"</strong>.
  </Prop>

    <a href="https://mui.com/api/button/#props" target="_blank">Material-UI <code> variant </code></a> property <strong>defaulted to "contained"</strong>.
  </Prop>

  <Prop icon="mui.png" name="children" type="node">
    Content of the <code>{' <Button /> '}</code>
  </Prop>

  <Prop icon="mui.png" name="{...props}" type="object">
